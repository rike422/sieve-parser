def self.parse_all(text)
  lines = text.split("\n")
  actions = []
  lines.each do |line|
    actions << self.new(line)
  end
  actions
end

def to_s
  text =""
  {'type'=>@type, 'copy'=>@copy, 'target'=>@target}.each do |name,item|
    if ['target'].index(name)
      text += "\"#{item}\" " unless item.nil?
    else
      text += "#{item} " unless item.nil?
    end

  end
  text[text.length-1] = ";"
  text
end
