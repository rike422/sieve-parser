def to_s
  text = "# #{name}\n"
  text += "#{@type}"
  if conditions.count > 1
    text += " #{@join} (" + conditions.join(", ") + ")"
  else
    text += " " + conditions[0].to_s
  end
  text += "\n{\n    "
  text += actions.join("\n    ")
  text += "\n}\n"
end
